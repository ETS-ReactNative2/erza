query Episodes($from: String, $to: String, $isAdult: Boolean, $paging: Paging, $ratings: [Int]) {
  episodes(fromDate: $from, toDate: $to, isAdult: $isAdult, paging: $paging, ratings: $ratings) {
    total
    hasMore
    nodes {
      id
      date
      rating
      note
      episode
      anime {
        id
        title
      }
    }
  }
}
